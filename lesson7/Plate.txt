package geekbrains.java1.lesson7;

public class Plate {
    private int food;

    public Plate(int food){
        this.food = food;
    }
    public void decreaseFood(int n,int m){
        if(food < n){
            System.out.println("Недостаточное количество еды, добавляем еду...");
            food += 100;
            food -= n;
            m += n;
            System.out.println("Еды хватает. Кот поел, сытость равна " + m + ". Кот наелся.");
        } else if(food >= n){
            food -= n;
            m += n;
            System.out.println("Еды хватает. Кот поел, сытость равна " + m + ". Кот наелся.");
        }
    }
    public void info(){
        System.out.println("Количество еды = " + food + ".");
    }
}
